# linux 安装

# 下载

https://www.boost.org/users/download/ 下载对应的unix版本，自己选择所需要的版本

# 安装，以1.70版本示例

tar -jxvf boost_1_70_0.tar.bz2
cd boost_1_70_0
./bootstrap.sh
sudo ./b2 install

# 测试(测试多线程的原子操作)

#include <boost/thread/thread.hpp>
#include <atomic>
#include <iostream>
#include <time.h>
#include <mutex>
using namespace std;

/* Before C++11 we can only do this
long total = 0;
mutex m;
void click(){
  for(int i =0; i<10000000;++i){
    m.lock();
    total += 1;
    m.unlock();
  }
}
*/

atomic_long total(0);
void click(){
  for(int i =0; i<10000000;++i){
    total += 1;
  }
}
int main(int argc, char** argv) {
  clock_t start = clock();
  boost::thread_group threads;
  for(int i = 0;i < 100;++i){
    threads.create_thread(click);
  }
  threads.join_all();
  clock_t finish = clock();
  cout<<"result:"<<total<<endl;
  cout<<"duration:"<<(finish - start)/CLOCKS_PER_SEC <<" s"<<endl;
  return 0;
}

  g++ click.cpp -lboost_thread -lpthread -o click
  ./click
