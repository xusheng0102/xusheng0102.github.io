# 最简单好用的exceljs

  1. 安装 npm i exceljs --save
  2. 新建一个excel.js
  const Excel = require('exceljs')
  let data = [{
    name: 'xxx',
    gender: 'male',
    age: '11'
  },{
    name: 'xxx1',
    gender: 'male',
    age: '11'
  },{
    name: 'xxx2',
    gender: 'male',
    age: '11'
  },{
    name: 'xxx3',
    gender: 'male',
    age: '11'
  }];
  
  const headers = [
    { header: '姓名', key: 'name', width: 8 },
    { header: '性别', key: 'gender', width: 12 },
    { header: '年龄', key: 'age', width: 25 } 
  ]
  // 下面这部分可以写成个function ,eg: function exportExcel(headers,data){...}
  const workbook = new Excel.Workbook(); // 创建Workbook对象
  const worksheet = workbook.addWorksheet('My Sheet'); // 创建一个worksheet并命名
  worksheet.columns = headers; // 设置列表属性

  worksheet.addRows(data); // 写入json数组数据
  worksheet.eachRow(row=>{
    row.height = 30;
  }); // 设置每行的高度，也可以单独设置 const row = worksheet.getRow(rownumber); row.heiget = 30;
  workbook.xlsx.writeFile('helloworld.xlsx')
    .then(function() {
        console.log('write ok')
  }); // 写入文件，writeFile的参数可以使文件名。可以写入到stream里等等。
  1. node excel.js 查看结果
